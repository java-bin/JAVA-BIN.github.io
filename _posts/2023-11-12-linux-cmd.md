## 생각날 때 마다 추가하는 Linux 명령어

```shell
# 뒤로이동(횟수에 따라)
$ cd ../(../../)
# 파일 안으로 이동 
$ cd [fileName]

# list 조회
$ ls/ll

# 파일 복사
$ cp [fileName] [directory]

# 파일 (강제) 삭제
$ rm (-rf) [fileName]

# 파일 이동
$ mv [fileName] [directory]

# 디렉토리 만들기
$ mkdir [fileName]

# 수정 날짜를 현재 시간으로 변경
$ touch [fileName]

# 파일 내용을 출력해 줌
$ cat [fileName]

# 파일을 수정할 수 있게 해 줌
$ vi/vim [fileName]
  $ insert / i : modify mode
  $ esc : modify mode 나가기
  $ :q : quit
  $ :q! : force quit
  $ :w : save
  $ :wq [fileName] : save and quit
  
  # search
  $ /[word] : 앞에서 부터 찾음
  $ ?[word] : 뒤에서 부터 찾음
  
  # change
  $ :%s/[word]/[word] : 첫번째 단어를 두번째 단어로 변경
  $ :%s/[word]/[word]/g : 모든 첫번쨰 단어를 두번째 단어로 변경
  $ :%s/[word]/[word]/gc : 바꾸기 전에 확인 후 모든 첫번쨰 단어를 두번째 단어로 변경
  
# file log (실시간)
$ tail -f [fileName]

# pipe / 필터링 해 
$ [명령어] | grep [word]

# 프로세스 확인
$ lsof -i :[portNumber]

# 프로세스 강제 종료
$ kill -9 [processPID]
  

```
